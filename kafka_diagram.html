1. Producer Example (‡∏™‡πà‡∏á Message)

package main

import (
    "context"
    "fmt"
    "log"
    "time"

    "github.com/segmentio/kafka-go"
)

func main() {
    // ‡∏™‡∏£‡πâ‡∏≤‡∏á Producer (Writer)
    writer := kafka.NewWriter(kafka.WriterConfig{
        Brokers:  []string{"localhost:9092"},
        Topic:    "orders",
        Balancer: &kafka.LeastBytes{}, // ‡∏Å‡∏£‡∏∞‡∏à‡∏≤‡∏¢ message ‡πÑ‡∏õ partition
    })
    defer writer.Close()

    // ‡∏™‡πà‡∏á Message

        {
            Key:   []byte("order-1"), // Key ‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô‡πÑ‡∏õ‡∏Å partition ‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô
            Value: []byte("Order #1: iPhone 15"),
        },
        {
            Key:   []byte("order-2"),
            Value: []byte("Order #2: MacBook Pro"),
        },
        {
            Key:   []byte("order-3"),
            Value: []byte("Order #3: AirPods"),
        },
    }

    ctx := context.Background()
    
    for _, msg := range messages {
        err := writer.WriteMessages(ctx, msg)
        if err != nil {
            log.Fatal("failed to write message:", err)
        }
        fmt.Printf("‚úÖ Sent: %s = %s\n", msg.Key, msg.Value)
        time.Sleep(500 * time.Millisecond)
    }
}

2. Consumer Example (‡∏≠‡πà‡∏≤‡∏ô Message)

package main

import (
    "context"
    "fmt"
    "log"

    "github.com/segmentio/kafka-go"
)

func main() {
    // ‡∏™‡∏£‡πâ‡∏≤‡∏á Consumer (Reader)
    reader := kafka.NewReader(kafka.ReaderConfig{
        Brokers:  []string{"localhost:9092"},
        Topic:    "orders",
        GroupID:  "order-processor-group", // Consumer Group
        MinBytes: 10e3, // 10KB
        MaxBytes: 10e6, // 10MB
    })
    defer reader.Close()

    fmt.Println("üîç Starting consumer...")

    // ‡∏≠‡πà‡∏≤‡∏ô Message Loop
    for {
        msg, err := reader.ReadMessage(context.Background())
        if err != nil {
            log.Fatal("failed to read message:", err)

        // ‡πÅ‡∏™‡∏î‡∏á‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• Message
        fmt.Printf("üì® Message from Partition %d [Offset %d]: %s = %s\n",

        // Process message ‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì‡∏ó‡∏µ‡πà‡∏ô‡∏µ‡πà
        processOrder(msg)
        
        // Offset ‡∏à‡∏∞‡∏ñ‡∏π‡∏Å commit ‡∏≠‡∏±‡∏ï‡πÇ‡∏ô‡∏°‡∏±‡∏ï‡∏¥
    }
}

3. Manual Offset Commit (Commit ‡πÄ‡∏≠‡∏á)

package main

import (
    "context"
    "fmt"
    "log"

    "github.com/segmentio/kafka-go"
)

func main() {
    reader := kafka.NewReader(kafka.ReaderConfig{
        Brokers: []string{"localhost:9092"},
        Topic:   "orders",
        GroupID: "manual-commit-group",
    })
    defer reader.Close()

    for {
        msg, err := reader.FetchMessage(context.Background())
        if err != nil {
            log.Fatal(err)
        }

        fmt.Printf("üì® Received: %s\n", string(msg.Value))

        // ‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏• message
        err = processOrderSafely(msg)
        if err != nil {
            fmt.Printf("‚ùå Failed to process: %v\n", err)
            continue // ‡πÑ‡∏°‡πà commit ‡∏ñ‡πâ‡∏≤‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡∏•‡πâ‡∏°‡πÄ‡∏´‡∏•‡∏ß
        }

        // Commit offset ‡πÄ‡∏°‡∏∑‡πà‡∏≠‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à‡πÅ‡∏•‡πâ‡∏ß
        err = reader.CommitMessages(context.Background(), msg)
        if err != nil {
            fmt.Printf("‚ùå Failed to commit offset: %v\n", err)
        } else {
            fmt.Printf("‚úÖ Committed offset %d\n", msg.Offset)
        }
    }
}

func processOrderSafely(msg kafka.Message) error {
    // Business logic ‡∏ó‡∏µ‡πà‡∏≠‡∏≤‡∏à error ‡πÑ‡∏î‡πâ
    fmt.Printf("‚öôÔ∏è  Processing: %s\n", string(msg.Value))
    return nil // return error ‡∏ñ‡πâ‡∏≤‡∏°‡∏µ‡∏õ‡∏±‡∏ç‡∏´‡∏≤
}

4. Producer with Partition Key

package main

import (
    "context"
    "fmt"
    "log"

    "github.com/segmentio/kafka-go"
)

func main() {
    writer := kafka.NewWriter(kafka.WriterConfig{
        Brokers:  []string{"localhost:9092"},
        Topic:    "user-events",
        Balancer: &kafka.Hash{}, // ‡πÉ‡∏ä‡πâ Hash ‡∏Ç‡∏≠‡∏á Key ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÄ‡∏•‡∏∑‡∏≠‡∏Å Partition
    })
    defer writer.Close()

    // Message ‡∏ó‡∏µ‡πà‡∏°‡∏µ Key ‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô‡∏à‡∏∞‡πÑ‡∏õ‡∏Å Partition ‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô
    // = ‡∏£‡∏±‡∏ö‡∏õ‡∏£‡∏∞‡∏Å‡∏±‡∏ô Order ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö user ‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô
    users := []string{"user-123", "user-456", "user-123", "user-789"}
    events := []string{"login", "login", "purchase", "logout"}

    for i := 0; i < len(users); i++ {
        msg := kafka.Message{
            Key:   []byte(users[i]),
            Value: []byte(fmt.Sprintf("%s: %s", users[i], events[i])),
        }

        err := writer.WriteMessages(context.Background(), msg)
        if err != nil {
            log.Fatal(err)
        }
        
        fmt.Printf("‚úÖ Sent event for %s: %s\n", users[i], events[i])
    }
    
    fmt.Println("\nüí° user-123 events ‡∏à‡∏∞‡πÑ‡∏õ‡∏Å partition ‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏Å‡∏±‡∏ô = ‡∏£‡∏±‡∏ö‡∏õ‡∏£‡∏∞‡∏Å‡∏±‡∏ô order")
}
